
CREATE TABLE benutzer (
    id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    benutzername VARCHAR(30) NOT NULL,
    passwort VARCHAR(255) NOT NULL
);



CREATE TABLE lieder (
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(255),
    autor VARCHAR(255),
    ton VARCHAR(255),
    pdf_attachment BLOB,
    hinzugefuegt_am DATETIME,
    PRIMARY KEY (id)
);
